Widgets in iFrames

Leaflet 01

library(leaflet)
library(widgetframe)
## Loading required package: htmlwidgets
l <- leaflet() %>% addTiles() %>% setView(0,0,2)
frameWidget(l, height = '400')

Leaflet 02

library(leaflet)
suppressPackageStartupMessages(library(widgetframe))
l <- leaflet() %>% addProviderTiles(providers$CartoDB.DarkMatter) %>% setView(0,0,2)
frameWidget(l, height = '300')

Dygraph

library(dygraphs)
ts <- dygraph(nhtemp, main = "New Haven Temperatures")
frameWidget(ts, height = 350, width = '95%')

DT

library(DT)
dt <-  datatable(
  head(iris, 20), 
  options = list(
     columnDefs = list(list(className = 'dt-center', targets = 5)),
     pageLength = 5, lengthMenu = c(5, 10, 15, 20)),
  fillContainer = T)

frameWidget(dt, height = 350, width = '95%')
