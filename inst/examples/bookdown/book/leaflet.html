1 Leaflet

1.1 Example 1

library(leaflet)
library(widgetframe)
## Loading required package: htmlwidgets
l <- leaflet() %>% addTiles() %>% setView(0,0,2)
frameWidget(l, height = '400')

1.2 Example 2

l <- leaflet() %>%
  addProviderTiles(providers$CartoDB.DarkMatter) %>%
  setView(0,0,2)
frameWidget(l, height = '400')
